# 小程序任务消息通知与执行流程 · 实现方案

> **For agentic workers:** REQUIRED SUB-SKILL: Use `subagent-driven-development` (recommended) or `executing-plans` to implement this plan task-by-task.

**Goal:** 在小程序侧实现统一的任务消息通知入口和执行流程，覆盖探盘打卡、话术训练、内容发布三种任务类型。

**Architecture:** 
- 首页 tools.html 顶部增加通知铃铛入口
- 通知面板采用底部弹窗形式（不跳转页面）
- 探盘任务复用 task-detail.html，新增4阶段执行流程
- 话术训练跳转 speech-train.html
- 内容发布根据状态显示提示或跳转

**Tech Stack:** HTML/CSS/JavaScript (小程序原型)，TailwindCSS

### 任务3: 实现通知面板交互逻辑

**Files:**
- Modify: `miniprogram/tools.html`

**Step 1: 添加面板控制函数**

在 `<script>` 末尾添加：

```javascript
// 通知面板显示/隐藏
function showNotificationPanel() {
    document.getElementById('notifyOverlay').classList.add('show');
    document.getElementById('notifyPanel').classList.add('show');
    renderNotificationList('all');
}

function hideNotificationPanel() {
    document.getElementById('notifyOverlay').classList.remove('show');
    document.getElementById('notifyPanel').classList.remove('show');
}

// 渲染通知列表
function renderNotificationList(filterType) {
    const list = document.getElementById('notifyList');
    const data = window.notificationData || [];
    
    const filtered = filterType === 'all' 
        ? data 
        : data.filter(item => item.taskType === filterType);
    
    if (filtered.length === 0) {
        list.innerHTML = '<div class="p-8 text-center text-gray-400">暂无通知</div>';
        return;
    }
    
    list.innerHTML = filtered.map(item => `
        <div class="notification-item ${item.isRead ? 'read' : ''}" 
             onclick="handleNotificationClick(${item.id}, '${item.taskType}')">
            <div class="flex items-center gap-2 mb-1">
                <span class="${item.isRead ? 'bg-gray-300' : 'bg-red-500'} w-2 h-2 rounded-full"></span>
                <span class="font-medium text-gray-800">${item.taskName}</span>
            </div>
            <div class="text-sm text-gray-500 mb-1">${item.message}</div>
            <div class="text-xs text-gray-400">${formatTime(item.createTime)}</div>
        </div>
    `).join('');
}

// 分类筛选
function filterNotifications(type) {
    // 更新Tab状态
    document.querySelectorAll('.filter-tab').forEach(tab => {
        tab.classList.remove('active');
        if (tab.dataset.type === type) tab.classList.add('active');
    });
    renderNotificationList(type);
}

// 点击通知
function handleNotificationClick(id, type) {
    // 标记已读
    const item = window.notificationData.find(n => n.id === id);
    if (item) item.isRead = true;
    updateBadge();
    
    hideNotificationPanel();
    
    // 跳转对应页面
    if (type === 'exploration' || type === 'speech' || type === 'content') {
        window.location.href = 'calendar.html';
    }
}

// 清空全部
function clearAllNotifications() {
    window.notificationData = [];
    updateBadge();
    renderNotificationList('all');
}

// 更新红点数字
function updateBadge() {
    const badge = document.getElementById('notifyBadge');
    const unreadCount = (window.notificationData || []).filter(n => !n.isRead).length;
    badge.textContent = unreadCount;
    badge.style.display = unreadCount > 0 ? 'flex' : 'none';
}

// 格式化时间
function formatTime(date) {
    const now = new Date();
    const diff = now - date;
    const minutes = Math.floor(diff / 60000);
    const hours = Math.floor(diff / 3600000);
    const days = Math.floor(diff / 86400000);
    
    if (minutes < 1) return '刚刚';
    if (minutes < 60) return minutes + '分钟前';
    if (hours < 24) return hours + '小时前';
    if (days < 7) return days + '天前';
    return date.toLocaleDateString();
}

// 页面加载时初始化
document.addEventListener('DOMContentLoaded', updateBadge);
```
