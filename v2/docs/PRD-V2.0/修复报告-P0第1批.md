```javascript
function showModal(type, title, message, duration) {
  // type: 'success' | 'error' | 'info' | 'warning'
  var iconMap = {
    success: '<i class="fa fa-check-circle"></i>',
    error: '<i class="fa fa-times-circle"></i>',
    info: '<i class="fa fa-info-circle"></i>',
    warning: '<i class="fa fa-exclamation-circle"></i>'
  };
  document.getElementById('globalModalIcon').innerHTML = iconMap[type] || iconMap.info;
  document.getElementById('globalModalTitle').textContent = title;
  document.getElementById('globalModalBody').textContent = message;
  document.getElementById('globalModalFooter').innerHTML =
    '<button class="btn-primary" onclick="closeGlobalModal()">知道了</button>';
  document.getElementById('globalModalOverlay').classList.add('active');
  document.body.style.overflow = 'hidden';
  if (duration !== false) {
    setTimeout(closeGlobalModal, duration || 2500);
  }
}
```

**confirmModal() 函数（用于替代 confirm()）：**
```javascript
function confirmModal(title, message, onConfirm) {
  document.getElementById('globalModalIcon').innerHTML = '<i class="fa fa-question-circle"></i>';
  document.getElementById('globalModalIcon').className = 'global-modal-icon confirm';
  document.getElementById('globalModalTitle').textContent = title;
  document.getElementById('globalModalBody').textContent = message;
  document.getElementById('globalModalFooter').innerHTML =
    '<button class="btn-cancel" onclick="closeGlobalModal()">取消</button>' +
    '<button class="btn-primary" onclick="closeGlobalModal();window._doConfirm()">确认</button>';
  document.getElementById('globalModalOverlay').classList.add('active');
  document.body.style.overflow = 'hidden';
  window._confirmCallback = onConfirm || null;
}
function _doConfirm() {
  if (typeof window._confirmCallback === 'function') window._confirmCallback();
  window._confirmCallback = null;
}
```

**showToast() 函数：**
```javascript
function showToast(message, type, duration) {
  var toast = document.getElementById('globalToast');
  if (!toast) return;
  toast.className = 'toast-notification ' + (type || '');
  toast.textContent = message;
  toast.classList.add('show');
  clearTimeout(toast._timer);
  toast._timer = setTimeout(function() {
    toast.classList.remove('show');
  }, duration || 2200);
}
```
